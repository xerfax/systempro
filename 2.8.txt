Создайте строковую переменную для хранения пароля, необходимого для доступа к секретному сообщению. Пользователь должен ввести пароль, и программа проверит его на правильность. Если введённый пароль неверный, пользователю будет предложено ввести пароль снова. Если пароль правильный, отобразите секретное сообщение.
Если пользователь не сможет ввести правильный пароль после трёх попыток, программа должна завершиться.

using System;

class Program
{
    static void Main()
    {
        // Определяем правильный пароль
        const string correctPassword = "mypassword"; // Замените на ваш пароль
        const int maxAttempts = 3; // Максимальное количество попыток

        int attemptCount = 0; // Счетчик попыток
        bool isPasswordCorrect = false; // Флаг для проверки правильности пароля

        // Цикл для проверки пароля
        while (attemptCount < maxAttempts && !isPasswordCorrect)
        {
            Console.Write("Введите пароль для доступа к секретному сообщению: ");
            string userInput = Console.ReadLine(); // Ввод пароля пользователем

            if (userInput == correctPassword)
            {
                isPasswordCorrect = true; // Устанавливаем флаг, если пароль правильный
            }
            else
            {
                attemptCount++; // Увеличиваем счетчик попыток
                Console.WriteLine($"Неверный пароль. У вас осталось {maxAttempts - attemptCount} попыток.");
            }
        }

        // Проверяем, был ли введен правильный пароль
        if (isPasswordCorrect)
        {
            // Вывод секретного сообщения
            Console.WriteLine("Секретное сообщение: Вы успешно получили доступ к секрету!");
        }
        else
        {
            // Уведомление о завершении программы
            Console.WriteLine("Вы исчерпали все попытки. Доступ запрещен.");
        }
    }
}
