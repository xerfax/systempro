Легенда:
Вы пришли в магазин, чтобы обменять своё золото на кристаллы. У вас есть определённое количество золота, и продавец спрашивает, сколько кристаллов вы хотите приобрести. После покупки у вас останется часть золота и появятся кристаллы.
Формально:
При запуске программы пользователь вводит количество золота. Затем ему предлагается купить кристаллы по установленной цене N (задать в программе). Пользователь вводит количество кристаллов, и его золото конвертируется. Выведите остаток золота и количество кристаллов. Проверка на достаточность золота не требуется.

using System;

class Program
{
    static void Main()
    {
        // Установленная цена кристалла
        int crystalPrice = 10;

        // Ввод начального количества золота от пользователя
        Console.Write("Введите количество золота: ");
        int userGold = int.Parse(Console.ReadLine());

        // Ввод количества кристаллов, которые хочет купить пользователь
        Console.Write("Сколько кристаллов вы хотите приобрести? ");
        int crystalsToBuy = int.Parse(Console.ReadLine());

        // Расчёт затрат золота и остатка
        int totalCost = crystalsToBuy * crystalPrice;      // Сумма, которую нужно заплатить за кристаллы
        int remainingGold = userGold - totalCost;          // Оставшееся золото после покупки

        // Вывод результатов
        Console.WriteLine($"Вы приобрели {crystalsToBuy} кристаллов.");
        Console.WriteLine($"Остаток золота: {remainingGold}");
    }
}
