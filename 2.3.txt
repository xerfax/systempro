// Создайте программу, которая с использованием циклов выводит последовательность: 5, 12, 19, 26, 33, 40, 47, 54, 61, 68, 75, 82, 89, 96, 103. Объясните, почему выбрали именно этот цикл. Используйте переменные для 
// обозначения чисел в условии цикла. Подсчитывать количество итераций не нужно. Даже если максимальное число изменится на 789, в коде нужно будет изменить только это значение.

using System;

class Program
{
    static void Main()
    {
        int startValue = 5;         
        int step = 7;               
        int maxValue = 103;         

        for (int currentValue = startValue; currentValue <= maxValue; currentValue += step)
        {
            Console.Write(currentValue);
            if (currentValue < maxValue)
            {
                Console.Write(", ");
            }
        }
        Console.WriteLine(); 
    }
}

// Цикл for - используется, поскольку он позволяет удобно управлять начальным значением, условием завершения и шагом. Условие завершения currentValue <= maxValue четко указывает, что цикл будет выполняться до достижения или превышения максимального значения.
