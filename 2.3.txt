// Создайте программу, которая с использованием циклов выводит последовательность: 5, 12, 19, 26, 33, 40, 47, 54, 61, 68, 75, 82, 89, 96, 103. Объясните, почему выбрали именно этот цикл. Используйте переменные для обозначения чисел в условии цикла. Подсчитывать количество итераций не нужно. Даже если максимальное число изменится на 789, в коде нужно будет изменить только это значение.

using System;

class Program
{
    static void Main()
    {
        // Задаем начальные значения
        int startValue = 5;         // Начальное значение
        int step = 7;               // Шаг последовательности
        int maxValue = 103;         // Максимальное значение для остановки

        // Цикл для вывода последовательности
        for (int currentValue = startValue; currentValue <= maxValue; currentValue += step)
        {
            Console.Write(currentValue);

            // Добавляем запятую после каждого числа, кроме последнего
            if (currentValue < maxValue)
            {
                Console.Write(", ");
            }
        }
        
        Console.WriteLine(); // Переход на новую строку после завершения вывода
    }
}

// Цикл for - используется, поскольку он позволяет удобно управлять начальным значением, условием завершения и шагом. Условие завершения currentValue <= maxValue четко указывает, что цикл будет выполняться до достижения или превышения максимального значения.
