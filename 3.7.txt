Вам дана строка с текстом. Используя метод String.Split(), разделите строку на слова, которые разделены пробелами. После этого выведите каждое слово на отдельной строке.
Пример:

 "Привет, мир!"Привет,
мир!

using System;

class Program
{
    static void Main()
    {
        // Инициализация строки с текстом
        string text = "Привет, мир!";
        
        // Разделение строки на слова, используя пробел в качестве разделителя
        string[] words = text.Split(' ');

        // Вывод каждого слова на новой строке
        foreach (string word in words)
        {
            Console.WriteLine(word);
        }
    }
}
